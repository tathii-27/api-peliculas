```powershell
# setup-backend.ps1
# Script para crear carpetas y generar un server.js vÃ¡lido

# --- Carpetas principales ---
$folders = @("backend", "config", "controllers", "middleware", "models", "routes")

foreach ($folder in $folders) {
    $path = Join-Path -Path "." -ChildPath $folder
    if (!(Test-Path $path)) {
        New-Item -ItemType Directory -Path $path | Out-Null
        Write-Host "Carpeta creada: $path"
    } else {
        Write-Host "Carpeta ya existe: $path"
    }
}

# --- Contenido JS como Here-String ---
$serverCode = @'
import express from "express";
import dotenv from "dotenv";

dotenv.config();
const app = express();
app.use(express.json());

// Ruta de prueba
app.get("/", (req, res) => {
  res.send("Servidor backend funcionando");
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Servidor escuchando en puerto ${PORT}`);
});
'@

# Guardar en backend/server.js
$serverPath = ".\backend\server.js"
Set-Content -Path $serverPath -Value $serverCode -Encoding UTF8
Write-Host "Archivo creado: $serverPath"

Write-Host "Setup inicial completado."
```
